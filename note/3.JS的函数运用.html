function toBlue () {
        document.getElementById('demo').style.width='400px';
        document.getElementById('demo').style.height='400px';
        document.getElementById('demo').style.background='blue';
      }
      function toRed () {
        document.getElementById('demo').style.width='200px';
        document.getElementById('demo').style.height='200px';
        document.getElementById('demo').style.background='red';
      }

 ,;() 

      这些符号不能少！！！！很多报错都是因为这些格式引起的。
